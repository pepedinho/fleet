28    pub async fn new(path: &std::path::Path) -> anyhow::Result<Self> {
29        let file = tokio::fs::OpenOptions::new()

36            file: Arc::new(Mutex::new(file)),

58    pub async fn fetchn(id: &str, n: usize) -> anyhow::Result<Vec<String>> {
59        let path = Logger::path_by_id(id);
60
61        // VÃ©rifier que le fichier existe
62        if !tokio::fs::try_exists(&path).await? {
63            return Err(anyhow::anyhow!("Failed to find log file"));
64        }
65
66        let mut file = File::open(&path).await?;
67        let metadata = file.metadata().await?;
68        let file_size = metadata.len();
69
70        let mut buffer = vec![0; 8192];
71        let mut collected = Vec::new();
72        let mut carry = String::new();
73
74        let mut pos = file_size as i64;
75
76        while pos > 0 && collected.len() < n {
77            let read_size = buffer.len().min(pos as usize);
78            pos -= read_size as i64;
79
80            file.seek(SeekFrom::Start(pos as u64)).await?;
81
82            file.read_exact(&mut buffer[..read_size]).await?;
83
84            let chunk = String::from_utf8_lossy(&buffer[..read_size]);
85
86            let combined = format!("{chunk}{carry}");
87            let mut parts: Vec<&str> = combined.split('\n').collect();
88
89            carry = parts.remove(0).to_string();
90
91            for line in parts.into_iter().rev() {
92                if !line.is_empty() {
93                    collected.push(line.to_string());
94                    if collected.len() >= n {
95                        break;
96                    }
97                }
98            }
99        }
100
101        if !carry.is_empty() && collected.len() < n {
102            collected.push(carry);
103        }
104
105        collected.reverse();
106
107        Ok(collected)
108    }

120    fn paint_level(&self, level: &str) -> String {
121        if !self.color_enable {
122            return level.to_string();
123        }
124        match level {
125            "INFO" => format!("{BG_BLUE}{FG_BOLD_WHITE} {level} {RESET}"),
126            "WARNING" => format!("{BG_ORANGE}{FG_BOLD_WHITE} {level} {RESET}"),
127            "ERROR" => format!("{BG_RED}{FG_BOLD_WHITE} {level} {RESET}"),
128            "JOB START" => format!("{BG_GREEN}{FG_BOLD_WHITE} {level} {RESET}"),
129            "JOB END" => format!("{BG_MAGENTA}{FG_BOLD_WHITE} {level} {RESET}"),
130            _ => level.to_string(),
131        }
132    }

134    pub async fn log(&self, level: &str, msg: &str) -> anyhow::Result<()> {
135        let mut f = self.file.lock().await;
136        let now = Local::now();
137        let line = format!(
138            "[{}] {}: {}\n",
139            now.format("%Y-%m-%d %H:%M:%S"),
140            self.paint_level(level),
141            msg
142        );
143        f.write_all(line.as_bytes()).await?;
144        f.flush().await?;
145        Ok(())
146    }

168    pub async fn clean(&self) -> anyhow::Result<()> {
