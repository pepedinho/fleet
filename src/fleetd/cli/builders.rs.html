13/// Handles watch-related CLI commands by delegating to subfunctions
14/// that build the appropriate [`DaemonRequest`].
15pub async fn handle_watch(cli: &Cli) -> Result<()> {
16    let watch_req = build_watch_request(cli).await?;
17    send_watch_request(watch_req).await?;
18    Ok(())
19}
20
21/// Builds the appropriate [`DaemonRequest`] for the given CLI command.
22pub async fn build_watch_request(cli: &Cli) -> Result<DaemonRequest> {
23    match &cli.command {
24        Commands::Watch { branch } => build_add_watch_request(branch.clone()),
25        Commands::Ps { all } => Ok(DaemonRequest::ListWatches { all: *all }),
26        Commands::Logs { id_or_name, follow } => build_logs_request(id_or_name, *follow),
27        Commands::Stop { id } => Ok(DaemonRequest::StopWatch { id: id.to_string() }),
28        Commands::Up { id } => Ok(DaemonRequest::UpWatch { id: id.to_string() }),
29        Commands::Rm { id } => Ok(DaemonRequest::RmWatch { id: id.to_string() }),
30        Commands::Stats => {
31            display_stats_interface().await?;
32            Ok(DaemonRequest::None)
33        }
34        Commands::Run { id } => Ok(DaemonRequest::RunPipeline { id: id.clone() }),
35    }
36}

38/// Builds an [`AddWatch`] request after validating configuration.
39fn build_add_watch_request(branch_cli: Option<String>) -> Result<DaemonRequest> {
40    let config_path = Path::new("./fleet.yml");
41    if !config_path.exists() {
42        return Err(anyhow::anyhow!(
43            "File `fleet.yml` missing from current directory."
44        ));
45    }
46
47    let config = load_config(config_path)?;
48    let branch = branch_cli
49        .or(config.branch.clone())
50        .unwrap_or(Repo::build(None)?.branch.clone());
51    let repo = Repo::build(Some(branch.clone()))?;
52
53    Ok(DaemonRequest::AddWatch {
54        project_dir: std::env::current_dir()?.to_string_lossy().into_owned(),
55        branch,
56        repo: Box::new(repo),
57        config: Box::new(config),
58    })
59}

61/// Builds a [`LogsWatches`] request from CLI or repository defaults.
62fn build_logs_request(id_or_name: &Option<String>, follow: bool) -> Result<DaemonRequest> {
63    match id_or_name {
64        Some(s) => Ok(DaemonRequest::LogsWatches {
65            id: s.to_string(),
66            f: follow,
67        }),
68        None => {
69            let repo = Repo::build(None)?;
70            Ok(DaemonRequest::LogsWatches {
71                id: repo.name,
72                f: follow,
73            })
74        }
75    }
